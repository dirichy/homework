\begin{problem}\label{pro:1}
  设\(Z \)是随机变量，\(F \)是其分布函数。证明：
  \begin{itemize}
    \item \(\mathbb{E}(Z^+) = \int_{0}^{\infty}(1 - F(x)) \d x \)。
    \item \(\mathbb{E}(Z^-) = \int_{-\infty}^{0} F(x) \d x \)。
  \end{itemize}
\end{problem}
\begin{solution}
  由 Fubini 定理知：
  \(\mathbb{E}(Z^+) = \mathbb{E}(\int_{0}^{\infty} \mathbbm{1}_{[0,Z^+)}(x) \d x) = \int_{0}^{\infty} \mathbb{E}(\mathbbm{1}_{[0,Z^+)}(x)) \d x = \int_{0}^{\infty}(1-F(x)) \d x \)。
  同理，有
  \(\mathbb{E}(Z^-) = \mathbb{E}(\int_{-\infty}^{0} \mathbbm{1}_{(-Z^-,0]}(x) \d x) = \int_{-\infty}^{0}\mathbb{E}(\mathbbm{1}_{(-Z^-,0]}(x)) \d x = \int_{-\infty}^{0}F(x) \d x \)。
\end{solution}
