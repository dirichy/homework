\begin{problem}\label{pro:1}
Find an $n-$dimensional real matrix representation of $(\R,+)$. 
\end{problem}


\begin{problem}\label{pro:2}
Find an infinitely dimensional representation of $(\R,+)$. 
\end{problem}

\begin{problem}
Determine whether the representation is faithful or not in 
Example 1~4 and 
\Cref{pro:1},
\Cref{pro:2}.
\end{problem}


\begin{problem}
Let $\lambda\in \C^*$, and:
\begin{equation}
\phi_{\lambda}:(\Z,+)\to\C^*,n\mapsto \lambda^n
\end{equation}
Prove that $\phi_\lambda$ is an $1-$dimensional complex representation of $(\Z,+)$, and find when it's faithful.   
\end{problem}


\begin{problem}
Let $V=\mathbb{R}[x]$. $\forall a \in \mathbb{R}$, let:
$$
\begin{aligned}
& L_a(f(x)):=f(a x), \quad \forall f(x) \in \mathbb{R}[x], \\
& S_a(f(x)):=f\left(\mathrm{e}^a x\right), \quad \forall f(x) \in \mathbb{R}[x] .
\end{aligned}
$$
and:
$$
\begin{array}{ll}
\varphi(a)=L_a, & \forall a \in \mathbb{R}, \\
\psi(a)=S_a, & \forall a \in \mathbb{R} .
\end{array}
$$
Question: Is $\varphi$ and $\psi$ is infinitely dimensional real representation of $(\mathbb{R},+)$ ?
\end{problem}


\begin{problem}
Give the matrix representation of $G=\<a\>$ given by the regular representation, $\rho$, of field $K$, where $\rank(a)=4$. 
\end{problem}
