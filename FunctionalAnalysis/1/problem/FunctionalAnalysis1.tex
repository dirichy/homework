\begin{problem}
设 $(X, \d)$ 是距离空间, 令 $\rho(x, y)=\frac{\d(x, y)}{1+\d(x, y)}$. 求证 $(X, \rho)$ 也是距离空间. 
\end{problem}

\newcommand{\intt}[2]{\int_0^1|#1-#2| \mathrm{d} x}
\begin{problem}
$[0,1]$ 上的全体多项式记为$P[0,1]$, 定义距离
\begin{equation}
\d(p, q)=\int_0^1|p(x)-q(x)| \mathrm{d} x 
\end{equation}
其中 p, q \text { 是多项式 }. 证明
$(P[0,1],\d)$是不完备的, 并指出它的完备化空间.
\end{problem}
