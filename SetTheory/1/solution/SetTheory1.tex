\begin{problem}
Show that $\varnothing$ exists and is unique.
\end{problem}
\begin{solution}
Firstly, from axiom 0 we get $\exists x(x=x)$, then let $\phi(u)=u\hneq u$, from axiom 2 we get $\exists y\forall u(x\hin y\oto u\hin x\xand\phi(u))$. Obviously $\forall u(u\heq u)$, so we get $\forall u(\neg \phi(u))$, and then we get $\forall u\neg (u\hin y)$. i.e., $y=\emptyset$. 

On the other side, consider another emptyset $z$, we get $u\in y\iff u\neq u\iff u\in z$, then we get $u\hin y\oto u\hin z$, and then from axiom 1 we get $y\heq z$. So emptyset is unique. 
\end{solution}
\begin{problem}
Verify that
$$
\forall x \forall y \forall x^{\prime} \forall y^{\prime}\left((x, y)=\left(x^{\prime}, y^{\prime}\right) \rightarrow x=x^{\prime} \wedge y=y^{\prime}\right) .
$$
By induction, ordered $n$-tuples can be defined as follows:
$$
\left(x_1, \ldots, x_{n+1}\right)=\left(\left(x_1, \ldots, x_n\right), x_{n+1}\right) .
$$
Show that two ordered $n$-tuples $\left(x_1, \ldots, x_n\right)$ and $\left(y_1, \ldots, y_n\right)$ are equal iff $x_i=y_i$ for all $i=1, \ldots, n$.
\end{problem}
\begin{solution}
First from axiom 1 we easily get $x\heq y\oto \{x\}\heq \{y\}$.

Then consider $\{x\}\heq\{y,z\}$, form axiom 1 we get $y\hin\{x\}$, so $y\heq x$, and so do $z$. So we get $\{x\}\heq\{y,z\}\oto x\heq y\heq z$. 

Then we consider $\{a,b\}\heq \{c,d\}$. If $a\heq b$ then we get $\{a\}=\{c,d\}$, and thus $a\heq b\heq c \heq d$. If not, from axiom 1 we can get $a\heq c\xor a\heq d$. If $a\heq c$ we can easily get $b\heq d$, and if $a\heq d$ we can get $b\heq c$. So finially we get $\{a,b\}\heq \{c,d\}\oto (a\heq c\xand b\heq d)\xor(a\heq d\xand b\heq c)$. 

Now consider $(a,b)\heq (c,d)$, form the definition we get $\{\{a\},\{a,b\}\}\heq\{\{c\},\{c,d\}\}$, so we get $(\{a\}\heq\{c\}\xand \{a,b\}\heq \{c,d\})\xor (\{a\}\heq \{c,d\}\xand \{a,b\}\heq \{c\})$. 

If $\{a\}\heq \{c,d\}\xand \{a,b\}\heq \{c\})$, we can get $a\heq b\heq c\heq d$, so $a\heq c,b\heq d$. 

If $\{a\}\heq\{c\}\xand \{a,b\}\heq \{c,d\}$, we get $a\heq c$. If $a\heq b$, we get $a\heq b\heq c\heq d$. If $a\hneq b$, we get $c\hneq b$, but $b\hin \{c,d\}$, so $b\heq d$. 

So we finially proved $(a,b)\heq (c,d)\oto (a\heq c\xand b\heq d)$. 

Now we can use MI to proof $(x_1,\cdots x_n)=(y_1,\cdots y_n)\iff \band_{i=1}^n x_i=y_i$. 
Assume it's right for $k\leq n$, where $n\geq 2$, consider $n+1$. 
We have $(x_1,\cdots x_{n+1})=(y_1,\cdots y_{n+1})\iff (x_1,\cdots x_n)=(y_1,\cdots y_n)\xand x_{n+1}=y_{n+1}\iff x_i=y_i,\forall i=1,2,\cdots n+1$. 
\end{solution}
\begin{problem}
Show that if $R$ is a set, then $\operatorname{dom}(R)$ and $\operatorname{ran}(R)$ are sets. (In fact, field $(R)=\bigcup \bigcup R$).
\end{problem}
\begin{solution}
From axiom 2 we only need to prove $\field(R)\subset \cup\cup R$. i.e., $(u,v)\in R\to u\in \cup\cup R\xand v\in \cup\cup R$. 

Obviously we know $x\subset y\to \cup x\subset \cup y$. So $\{(u,v)\}\subset R\Rightarrow (u,v)\subset \cup R\Rightarrow \cup(u,v)\subset \cup\cup R$. Since $\cup(u,v)=\{u\}\cup\{u,v\}=\{u,v\}$, we get $u,v\in\cup(u,v)\subset \cup\cup R$, then $u,v\in\cup\cup R$. 
\end{solution}
\begin{problem}
Show that if $X, Y$ are sets then $\fun{X}{Y}$ is also a set.
\end{problem}
\begin{solution}
Firstly we prove that $\cup A\subset B\to A\subset \mi{B}$. 
Only need to prove $\forall x\in A,x\in \mi{B}$. Since $x\in\mi{B}\iff x\subset B$, we only need to prove $\forall u\in x,u\in B$. 
Noting $u\in x\xand x\in A\to u\in \cup A$ and $\cup A\subset B$, so $u\in x\xand x\in A\to u\in B$. 

Second we prove $A\times B:=\{(x,y):x\in A,y\in B\}$ exists. 
From axiom 2 we only need to prove $A\times B\subset \mi{\mi{A\cup B}}$. 
Since $\cup(A\times B)=\{\{x\}:x\in A\}\cup\{\{x,y\}:x\in A,y\in B\}$, we get $\cup\cup(A\times B)=A\cup B\subset A\cup B$, so $\cup(A\times B)\subset \mi{A\cup B}$, and thus $A\times B\subset\mi{\mi{A\cup B}}$. So from axiom 2 we get $A\times B$ exists. 

Now we can prove the existence of $\fun{X}{Y}$. Noting $\forall f\in \fun{X}{Y},f\subset X\times Y$, so $\fun{X}{Y}\subset \mi{X\times Y}$. From axiom 2 we know $\fun{X}{Y}$ exists. 
\end{solution}

\begin{problem}
    There is no set $X$ such that $\mathscr{P}(X) \subseteq X$.
\end{problem}
\begin{solution}
    If there is such $X$, we consider the set $Y:=\{x\in X:x\notin x\}\subset X$. If $Y\in Y$, then we get $Y\notin Y$. If $Y \notin Y$, since $Y\in \mi{X}\subset X$ we get $Y\in Y$. So it's a contradiction. So there is no such $X$. 
\end{solution}
\begin{problem}
    If $X$ is inductive, then the set
    $$
    \{x \in X \mid x \subseteq X\}
    $$
    is inductive. Hence $N$ is transitive, and for each $n$, $n=\{m \in N \mid m<n\}$.
\end{problem}
\begin{solution}
    Let $x\in X\xand x\subset X$. Since $X$ is inductive, we get $x\cup\{x\}\in X$. Since $x\in X$ we get $\{x\}\subset X$. So $x\cup\{x\}\subset X$. So $\{x\in X:x\subset X\}$ is inductive. 
    Obviously $\emptyset \in\{x\in X:x\subset X\}$, so by MI we can get $\N\subset \{x\in \N:x\subset \N\}$, so $ \N= \{x\in \N:x\subset \N\}$. i.e., $\N$ is transitive. 

    Since $\forall m,n\in \N(m<n\oto m\in n)$, so $\{m\in \N:m<n\}\subset n$. Since $\N$ is transitive, so $m\in n\to m\in \N$, so $n\subset \{m\in \N:m<n\}$. So finally we get $n=\{m\in \N:m<n\}$. 
\end{solution}
\begin{problem}
    If $X$ is inductive, then the set
    $$
    \{x \in X \mid x \text { is transitive }\}
    $$
    is inductive. Hence every $n \in N$ is transitive.
\end{problem}
\begin{solution}
    Use $\tau(x)$ to reperesent $x$ is transitive. Let $x\in X\xand \tau(x)$, consider $x\cup\{x\}$. Let $y\in x\cup \{x\}$, if $y=x$, then $y\subset x\cup \{x\}$; else, $y\in x$, since $x$ is transitive, so $y\subset x\subset x\cup\{x\}$. Then we get $\tau(x\cup\{x\})$. So $\{x\in X:\tau(x)\}$ is inductive. 

    For $\N$ we have $\{n\in\N:\tau(n)\}$ is inductive, and $0$ is transitive, so $\forall n\in\N,\tau(n)$.
\end{solution}

\begin{problem}
    If $X$ is inductive, then the set
    $$
    \{x \in X \mid x \text { is transitive and } x \notin x\}
    $$
    is inductive. Hence $n \notin n$ and $n \neq n+1$ for each $n \in N$
\end{problem}
\begin{solution}
    Use $\tau(x)$ to reperesent $x$ is transitive. Let $x\in X\xand \tau(x)\xand x\notin x$, consider $x\cup\{x\}$. Let $y\in x\cup \{x\}$, if $y=x$, then $y\subset x\cup \{x\}$; else, $y\in x$, since $x$ is transitive, so $y\subset x\subset x\cup\{x\}$. Then we get $\tau(x\cup\{x\})$. If $x\cup\{x\}\in x\cup\{x\}$, then $x\cup\{x\}\in x\xor x\cup\{x\}=x$. Since $x\notin x$ so $x\cup\{x\}\neq x$, so $x\cup\{x\}\in x$. Since $x$ is transitive, so $x\cup\{x\}\subset x$. But $x\notin x$, so it's impossible. 
    So $\{x\in X:\tau(x)\xand x\notin x\}$ is inductive. 

    For $\N$, we get $\{n\in \N:\tau(n)\xand x\notin x\}$ is inductive, so $\N=\{n\in \N:\tau(n)\xand x\notin x\}$, so $n\in\N\to n\notin n$. And since $n+1=n\cup\{n\}$, we get $n+1\neq n$. 
\end{solution}

\begin{problem}\label{pro:7}
    If $X$ is inductive, then the set
$\{x \in X \mid x$ is transitive and every nonempty
$$
z \subseteq x \text { has an } \in \text {-minimal element }\}
$$
is inductive.
( $t$ is $\in$-minimal in $z$ if there is no $s \in z$ such that $s \in t$.)
\end{problem}
\begin{solution}
    Use $\tau(x)$ to reperesent $x$ is transitive, use $\phi(x)$ to reperesent every nonempty $z\subset x$ has an $\in-$minimal element. Let $x\in X\xand \tau(x)\xand \phi(x)$, consider $x\cup\{x\}$. 
    If $x\in x$ then $x\cup\{x\}=x$ and thus $x\cup\{x\}\in \{y\in X:\tau(y)\xand \phi(y)\}$. Now we assume $x\notin x$. 

    Let $y\in x\cup \{x\}$, if $y=x$, then $y\subset x\cup \{x\}$; else, $y\in x$, since $x$ is transitive, so $y\subset x\subset x\cup\{x\}$. Then we get $\tau(x\cup\{x\})$. 
    Consider $z\subset x\cup\{x\}\xand z\neq \emptyset$. 
    If $z=\{x\}$, since $x\notin x$ so $x$ is the $\in-$minimal element of $z$. Else, consider $t=z\minus \{x\}\subset x$, and $u\in t$ is $\in-$minimal element of $t$. Since $u\in x\xand \tau(x)$ we get $u\subset x$. And $x\notin x\to x\notin u$. So $u$ is $\in-$minimal element of $t\cup\{x\}$, too. So $u$ is $\in-$minimal element of $z$.  
    So $\{x\in X:\tau(x)\xand \phi(x)\}$ is inductive. 
\end{solution}

\begin{problem}
    Every nonempty $X \subseteq \N$ has an $\in$-minimal element.
\end{problem}
\begin{solution}
    Consider $X\subset\N\xand X\neq \emptyset$. Assume $n\in X$. Consider $Y:=n\cap X\subset n$. If $Y=\emptyset$, then $n$ is the $\in-$minimal element of $X$. Else, from Problem \ref{pro:7} we know $Y$ has $\in-$minimal element $u$. 
    Now we just need to prove $u$ is $\in-$minimal element of $X$. 
    If not, $\exists m \in X,m\in u$, then since $u\in n$ we get $m\in u\subset n$, so $m\in n$ and thus $m\in Y$, contradiction with $u$ is $\in-$minimal element of $Y$. 
\end{solution}

\begin{problem}
    If $X$ is inductive then so is
$$
\{x \in X \mid x=\varnothing \vee x=y \cup\{y\} \text { for some } y\}. 
$$
Hence each $n \neq \varnothing$ is $m+1$ for some $m$.
\end{problem}
\begin{solution}
    Consider $x\in X\xand (x=\emptyset \xor (\exists z x=z\cup\{z\}))$, then for $y=x\cup\{x\}$ we have $y\in X\xand y=x\cup\{x\}$, thus $y\in\{x\in X:x=\emptyset\xor (\exists y\in X) x=y\cup\{y\}\}$. So the set is inductive, too. 

    For $\N$ we get $\forall n\in\N,n=\emptyset \xor \exists m\in \N,n=m\cup\{m\}$. If $n=\emptyset$ then $n=0$. If $n=m\cup\{m\}$ then $m\in n$. Since $\N$ is transitive, we get $m\in n\subset \N$. So $m\in \N$. So $n=m\cup\{m\}=m+1$. 
\end{solution}

\begin{problem}
    Let $A$ be a subset of $N$ such that $0 \in A$, and if $n \in A$ then $n+1 \in A$. Then $A=N$. 
\end{problem}
\begin{solution}
    Obviously $A$ is inductive, and $\N$ is the least inductive set, so $\N\subset A$. Noting $A\subset \N$, so $A=\N$. 
\end{solution}
